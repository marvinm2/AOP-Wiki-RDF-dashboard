        font=dict(family="Arial, sans-serif", size=13, color=BRAND_COLORS['primary']),
        paper_bgcolor="white",
        plot_bgcolor="white",
        hovermode="x unified",
        autosize=True,

            xanchor="left",
            x=1.02,
            font=dict(size=11),
        ),
        xaxis=dict(
            gridcolor="#e0e0e0",
            linecolor="#cccccc",
            zeroline=False,
            tickfont=dict(size=11),
        ),
        yaxis=dict(
            gridcolor="#e0e0e0",
            linecolor="#cccccc",
            zeroline=False,
            tickfont=dict(size=11),
        ),
    )
)

pio.templates["vhp4safety"] = vhp4safety_template
pio.templates.default = "plotly_white+vhp4safety"

4. **render_plot_html() helper** — replaces all individual `pio.to_html()` calls:
```python
def render_plot_html(fig, include_plotlyjs=False):
    """Render a Plotly figure to HTML with standardized config.

    Uses shared PLOTLY_HTML_CONFIG for consistent toolbar behavior
    (hidden until hover, no Plotly logo, responsive sizing).
    """
    return pio.to_html(
        fig,
        full_html=False,
        include_plotlyjs=include_plotlyjs,
        config=PLOTLY_HTML_CONFIG,
    )
```
